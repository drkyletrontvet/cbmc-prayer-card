// ── Submit ────────────────────────────────────────────────

function submitCard() {

  // Validate

  var nameSel    = document.getElementById('f-name').value;

  var recipSel   = document.getElementById('f-recipient').value;


  if (!nameSel) {

    showError('Please select your name before submitting.');

    return;

  }

  if (!recipSel) {

    showError('Please select a prayer partner to send this card to.');

    return;

  }


  var submitterName = (nameSel === '__guest__') ? gv('f-guest') : nameSel;

  if (!submitterName) {

    showError('Please enter your name in the Guest field.');

    return;

  }


  var submitterInfo = CONTACTS[nameSel] || {biz:'', phone:'', email:''};

  var recipientInfo = (recipSel !== '__guest__') ? (CONTACTS[recipSel] || {}) : {email:'', biz:''};


  var payload = {

    name:          submitterName,

    email:         submitterInfo.email,

    biz:           submitterInfo.biz,

    phone:         submitterInfo.phone,

    date:          gv('f-date'),

    personal:      gv('f-personal'),

    bizPrayer:     gv('f-biz'),

    paulTimothy:   gv('f-paultim'),

    unconvinced:   gv('f-unconvinced'),

    recipientName: recipSel === '__guest__' ? 'Friend' : recipSel,

    recipientEmail: recipientInfo.email || ''

  };


  if (!payload.recipientEmail) {

    showError('No email found for the selected prayer partner. Please contact your group leader.');

    return;

  }


  // Check if Apps Script URL is configured

  if (!SCRIPT_URL || SCRIPT_URL === 'PASTE_YOUR_APPS_SCRIPT_URL_HERE') {

    showError('The form is not yet connected to the backend. See CBMC_Setup_Guide.md to complete setup.');

    return;

  }


  // UI — loading state

  var btn = document.getElementById('btn-submit');

  btn.disabled = true;

  btn.innerHTML = '<span class="loading-spinner"></span>Sending...';

  hideMessages();


  fetch(SCRIPT_URL, {

    method:  'POST',

    headers: {'Content-Type': 'application/json'},

    body:    JSON.stringify(payload)

  })

  .then(function(res) { return res.json(); })

  .then(function(json) {

    btn.disabled = false;

    btn.innerHTML = 'Submit Prayer Card &amp; Send Email';

    if (json.status === 'success') {

      showSuccess(payload.recipientName, payload.recipientEmail);

      resetForm();

    } else {

      showError('Something went wrong: ' + (json.message || 'Unknown error.'));

    }

  })

  .catch(function(err) {

    btn.disabled = false;

    btn.innerHTML = 'Submit Prayer Card &amp; Send Email';

    showError('Could not reach the server. Check your internet connection and try again. (' + err.message + ')');

  });

}


function showSuccess(recipName, recipEmail) {

  var banner = document.getElementById('success-banner');

  document.getElementById('success-text').innerHTML =

    'Your prayer card was emailed to <strong>' + recipName + '</strong> (' + recipEmail + '). '

    + 'Two follow-up reminders have been automatically scheduled before the next CBMC meeting.';

  banner.style.display = 'block';

  document.getElementById('error-banner').style.display = 'none';

}


function showError(msg) {

  var banner = document.getElementById('error-banner');

  banner.textContent = '⚠ ' + msg;

  banner.style.display = 'block';

  document.getElementById('success-banner').style.display = 'none';

}


function hideMessages() {

  document.getElementById('success-banner').style.display = 'none';

  document.getElementById('error-banner').style.display = 'none';

}


function resetForm() {

  ['f-personal','f-biz','f-paultim'].forEach(function(id) {

    document.getElementById(id).value = '';

  });

  document.getElementById('f-unconvinced').value = '';

  document.getElementById('f-recipient').value = '';

}
